        def deliver!(headers, bodies)
      from_addr = @from_addr
      Mail.deliver(from_addr) do
        to headers[:to]
        from from_addr
        subject headers[:subject]
        text_part do
          body bodies[:txt]
        end
        html_part do
          body bodies[:html]
        end
      end
    end
